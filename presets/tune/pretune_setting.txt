#$ TITLE: Pre-Tune Setting
#$ FIRMWARE_VERSION: 4.5
#$ CATEGORY: TUNE
#$ STATUS: EXPERIMENTAL
#$ KEYWORDS: tune
#$ AUTHOR: Whistler2008
#$ DESCRIPTION: Pre-Tune Setting
#$ DESCRIPTION: 
#$ DESCRIPTION: WARNING! For RPM Filter, you must flash ESC firmware with RPM Filter support
#$ WARNING: EXPERIMENTAL

#--------------------------------------------------#

#$ OPTION_GROUP BEGIN: Basic Setting

	#$ OPTION BEGIN (UNCHECKED): Pilot Name
		### Pilot Name ###
		
		set pilot_name = TUNNING
		set craft_name = TUNNING PURPOSE
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): CAM Angle 5%
		### CAM Angle 5% ###
		
		set fpv_mix_degrees = 5
		set small_angle = 180
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): RC Smoothing 30
		### RC Smoothing 30 ###
		
		set rc_smoothing_auto_factor = 30
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): Beeper
		### Custom Beeper Setting ###

		beeper ACC_CALIBRATION
		beeper ACC_CALIBRATION_FAIL
		beeper -ARMED
		beeper ARMING
		beeper ARMING_GPS_FIX
		beeper ARMING_GPS_NO_FIX
		beeper BAT_CRIT_LOW
		beeper -BAT_LOW
		beeper BLACKBOX_ERASE
		beeper -CAM_CONNECTION_CLOSE
		beeper -CAM_CONNECTION_OPEN
		beeper CRASH_FLIP
		beeper DISARMING
		beeper -DISARM_REPEAT
		beeper GPS_STATUS
		beeper GYRO_CALIBRATED
		beeper MULTI_BEEPS
		beeper -ON_USB
		beeper READY_BEEP
		beeper -RX_LOST
		beeper RX_LOST_LANDING
		beeper RX_SET
		beeper SYSTEM_INIT
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): REVERSE / PROPS OUT Motor Direction
		### REVERSE Motor Direction ###
		
		set yaw_motors_reversed = ON
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): AVG Cell Telemetry Report
		### AVG Cell Telemetry Report ###
		
		set report_cell_voltage = ON
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): RPM Filter
		### Enable RPM Filter ###
		
		set motor_pwm_protocol = DSHOT300
		set dshot_bidir = ON
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): PreTune AUX
		### PreTune AUX ###

        # AUX
        aux 0 0 0 1550 2100 0 0
        aux 1 1 1 900 1300 0 0

        # Failsafe Setting
        set failsafe_procedure = AUTO-LAND
	#$ OPTION END

	#$ OPTION BEGIN (CHECKED): Tune BlackBox Setting
		### Blackbox for GyroFlow ###
		
		set blackbox_sample_rate = 1/2
		set debug_mode = GYRO_SCALED
		set blackbox_disable_pids = OFF
		set blackbox_disable_rc = OFF
		set blackbox_disable_setpoint = OFF
		set blackbox_disable_bat = OFF
		set blackbox_disable_alt = ON
		set blackbox_disable_rssi = ON
		set blackbox_disable_gyro = OFF
		set blackbox_disable_gyrounfilt = OFF
		set blackbox_disable_acc = OFF
		set blackbox_disable_debug = OFF
		set blackbox_disable_motors = OFF
		set blackbox_disable_rpm = OFF
		set blackbox_disable_gps = ON
		set blackbox_mode = NORMAL
		set blackbox_high_resolution = ON	
	#$ OPTION END

#$ OPTION_GROUP END

#--------------------------------------------------#

#$ OPTION_GROUP BEGIN: Tune Option

	#$ OPTION BEGIN (UNCHECKED): PID & Filter For Tune
		### PID & Rate For Filter Tune ###

		# master
		set gyro_lpf1_static_hz = 250
		set gyro_lpf2_static_hz = 500
		set gyro_lpf1_dyn_min_hz = 250
		set gyro_lpf1_dyn_max_hz = 500
		set simplified_gyro_filter_multiplier = 100

		simplified_tuning apply

		profile 0

		# profile 0
		set simplified_pids_mode = RPY
		set simplified_d_gain = 120
		set simplified_pi_gain = 100
		set simplified_feedforward_gain = 100
		set simplified_dmax_gain = 0
		set simplified_i_gain = 100
		set simplified_pitch_d_gain = 100
		set simplified_pitch_pi_gain = 100
		set simplified_master_multiplier = 100

		set dterm_lpf1_dyn_min_hz = 75
		set dterm_lpf1_dyn_max_hz = 150
		set dterm_lpf1_static_hz = 75
		set dterm_lpf2_static_hz = 150
		set simplified_dterm_filter_multiplier = 100

		set yaw_lowpass_hz = 100

		set angle_limit = 30

		simplified_tuning apply

		profile 1

		profile 2

		profile 3

		# restore original profile selection
		profile 0

	#$ OPTION END

	#$ OPTION BEGIN (UNCHECKED): Rateprofile For Tune
		### PID & Rate For PID Tune ###

		rateprofile 0

		# rateprofile 0
		set roll_rc_rate = 7
		set pitch_rc_rate = 7
		set yaw_rc_rate = 7
		set roll_expo = 0
		set pitch_expo = 0
		set yaw_expo = 0
		set roll_srate = 67
		set pitch_srate = 67
		set yaw_srate = 67

		rateprofile 1

		rateprofile 2

		rateprofile 3

		# restore original rateprofile selection
		rateprofile 0

	#$ OPTION END

#$ OPTION_GROUP END
